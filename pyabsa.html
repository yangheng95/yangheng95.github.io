Quick Start

Installation

pip install pyabsa

Basic Usage - Sentiment Classification

from pyabsa import ATEPCCheckpointManager

# Load a pretrained model
aspect_extractor = ATEPCCheckpointManager.get_aspect_extractor(
    checkpoint='english'
)

# Analyze sentiment
text = "The food was delicious but the service was slow."
result = aspect_extractor.extract_aspect(
    inference_source=[text],
    print_result=True
)

# Output:
# Aspect: food, Sentiment: Positive, Confidence: 0.98
# Aspect: service, Sentiment: Negative, Confidence: 0.95

Training Your Own Model

from pyabsa.tasks import AspectPolarityClassification as APC

# Configure training
config = APC.APCConfigManager.get_apc_config_english()
config.model = APC.APCModelList.BERT_SPC
config.num_epoch = 10
config.evaluate_begin = 5
config.max_seq_len = 128
config.pretrained_bert = 'bert-base-uncased'

# Train the model
trainer = APC.APCTrainer(
    config=config,
    dataset='laptop14',  # Built-in dataset
    auto_device=True
)

# Use the trained model
predictor = APC.APCPredictor(trainer.model, trainer.tokenizer)
result = predictor.predict("The battery life is amazing!")
